                  <h2 className="text-4xl font-black text-slate-900 tracking-tighter mb-6">Candidate Account Required</h2>
                  <p className="text-xl text-slate-500 font-medium mb-12 leading-relaxed">
                      You are currently logged in as a <span className="text-indigo-600 font-bold">{currentUser ? 'Corporate Client' : 'Recruitment Agent'}</span>. Only candidate profiles are authorized to express interest in job vacancies.
                  </p>
                  <div className="bg-blue-50 p-6 rounded-3xl border border-blue-100 text-left mb-12">
                      <div className="flex items-center gap-3 mb-2">
                          <Info size={18} className="text-blue-600" />
                          <p className="text-xs font-black uppercase text-blue-400 tracking-widest">Instruction</p>
                      </div>
                      <p className="text-sm font-bold text-slate-700 leading-snug">To apply for jobs, please sign out of your corporate account and create a personal <strong>Candidate Profile</strong>.</p>
                  </div>
                  <div className="flex flex-col gap-4">
                    <button onClick={handleSwitchToCandidate} className="w-full py-5 bg-indigo-600 text-white rounded-2xl font-black text-xs uppercase tracking-[0.2em] shadow-xl hover:bg-indigo-700 transition-all flex items-center justify-center gap-3">
                        <LogOut size={16}/> Switch to Candidate Profile
                    </button>
                    <button onClick={handleCloseModal} className="w-full py-5 bg-slate-900 text-white rounded-2xl font-black text-xs uppercase tracking-[0.2em] shadow-xl hover:bg-slate-700 transition-all">
                        I Understand
                    </button>
                  </div>
              </div>
          </div>
      )}

    </div>
  );
